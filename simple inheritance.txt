class Vehicle {
    String brand = "Ford"; 
    public void honk() {
        System.out.println("Beep beep!");
    }
}
class car extends Vehicle {
    String model = "Mustang";
    public void displayInfo() {
        System.out.println("Brand: " + brand);
        System.out.println("Model: " + model);
    }
}
public class Main {
    public static void main(String[] args) {
        car myCar = new car();
        myCar.honk();
        myCar.displayInfo();
    }
}
